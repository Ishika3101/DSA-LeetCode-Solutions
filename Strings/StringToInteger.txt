public class Atoi {
    public static void main(String[] args) {
        String s = "  -42";
        s = s.trim();

        int sign = 1, i = 0, result = 0;

        if (s.charAt(0) == '-' || s.charAt(0) == '+') {
            sign = (s.charAt(0) == '-') ? -1 : 1;
            i++;
        }

        while (i < s.length() && Character.isDigit(s.charAt(i))) {
            result = result * 10 + (s.charAt(i) - '0');
            i++;
        }

        System.out.println(result * sign);
    }
}
