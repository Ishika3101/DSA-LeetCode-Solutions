import java.util.*;

public class RemoveDuplicates {
    public static void main(String[] args) {
        String str = "programming";
        Set<Character> set = new LinkedHashSet<>();

        for(char ch : str.toCharArray()) {
            set.add(ch);
        }

        for(char ch : set) {
            System.out.print(ch);
        }
    }
}
