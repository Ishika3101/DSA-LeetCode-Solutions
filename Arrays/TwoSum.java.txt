import java.util.*;

int[] arr = {2,7,11,15};
int target = 9;

HashMap<Integer, Integer> map = new HashMap<>();

for (int i = 0; i < arr.length; i++) {
    int diff = target - arr[i];

    if (map.containsKey(diff)) {
        System.out.println(map.get(diff) + " " + i);
    }

    map.put(arr[i], i);
}
