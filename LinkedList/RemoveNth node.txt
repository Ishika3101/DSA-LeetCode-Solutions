public class RemoveNth {
    static class Node {
        int data;
        Node next;
        Node(int d) { data = d; }
    }

    public static Node removeNth(Node head, int n) {
        Node dummy = new Node(0);
        dummy.next = head;

        Node fast = dummy, slow = dummy;

        for (int i = 0; i <= n; i++)
            fast = fast.next;

        while (fast != null) {
            fast = fast.next;
            slow = slow.next;
        }

        slow.next = slow.next.next;
        return dummy.next;
    }
}
